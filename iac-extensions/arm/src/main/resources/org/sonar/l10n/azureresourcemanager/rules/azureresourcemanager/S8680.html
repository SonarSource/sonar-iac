Basic Authentication transmits a username and password (often base64-encoded, not encrypted) with every request. In Azure Logic Apps, using Basic auth for HTTP actions is insecure.

Why is this an issue?

  • Long-lived credentials -- passwords do not expire automatically and are difficult to rotate
  • No central revocation -- a compromised password cannot be invalidated from a single place
  • Replay attacks -- captured credentials can be reused indefinitely
  • Secrets in the definition -- the password ends up in the workflow definition, ARM exports, and run history

Azure Logic Apps should instead use Managed Identity or OAuth 2.0 / Entra ID (ActiveDirectoryOAuth).

Noncompliant Code Example

{
  "type": "Microsoft.Logic/workflows",
  "apiVersion": "2019-05-01",
  "name": "basic-auth-workflow",
  "properties": {
    "definition": {
      "actions": {
        "Call_API": {
          "type": "Http",
          "inputs": {
            "method": "GET",
            "uri": "https://api.example.com/data",
            "authentication": {
              "type": "Basic",
              "username": "apiuser",
              "password": "P@ssw0rd!"
            }
          }
        }
      },
      "triggers": {}
    }
  }
}

resource workflow 'Microsoft.Logic/workflows@2019-05-01' = {
  name: 'basic-auth-workflow'
  location: resourceGroup().location
  properties: {
    definition: {
      actions: {
        Call_API: {
          type: 'Http'
          inputs: {
            method: 'GET'
            uri: 'https://api.example.com/data'
            authentication: { // Noncompliant
              type: 'Basic'
              username: 'apiuser'
              password: 'P@ssw0rd!'
            }
          }
        }
      }
      triggers: {}
    }
  }
}

Compliant Solution

{
  "type": "Microsoft.Logic/workflows",
  "apiVersion": "2019-05-01",
  "name": "managed-identity-workflow",
  "identity": { "type": "SystemAssigned" },
  "properties": {
    "definition": {
      "actions": {
        "Call_API": {
          "type": "Http",
          "inputs": {
            "method": "GET",
            "uri": "https://api.example.com/data",
            "authentication": {
              "type": "ManagedServiceIdentity",
              "audience": "https://api.example.com"
            }
          }
        }
      },
      "triggers": {}
    }
  }
}

resource workflow 'Microsoft.Logic/workflows@2019-05-01' = {
  name: 'managed-identity-workflow'
  location: resourceGroup().location
  identity: { type: 'SystemAssigned' }
  properties: {
    definition: {
      actions: {
        Call_API: {
          type: 'Http'
          inputs: {
            method: 'GET'
            uri: 'https://api.example.com/data'
            authentication: {
              type: 'ManagedServiceIdentity'
              audience: 'https://api.example.com'
            }
          }
        }
      }
      triggers: {}
    }
  }
}
